import { Pipe, PipeTransform } from '@angular/core';
 
import { TranslateService } from 'ng2-translate';
import * as moment from 'moment';
 
 
import { Log, Level } from 'ng2-logger/ng2-logger';
const log = Log.create('shared/pipes/recentDatePipe');
 
const SECONDS_IN_DAY = 86400;
 
@Pipe({
    name: 'recentDate'
})
export class RecentDatePipe implements PipeTransform {
    constructor(private translateService: TranslateService) {
    }
 
    /**
     *
     * @param value timestamp
     * @param args date format string
     * @returns {string} 'today" | "yesterday" | <date formatted with *value* param>
     */
    transform(value: string, args: string): string {
        let todayMidnight     = moment().set({hour: 0, minute: 0, second: 0, millisecond: 0});
        let yesterdayMidnight = todayMidnight.clone().subtract(1, 'day');
        let date: moment.Moment;
        try {
            date = moment(value);
        } catch (e) {
            log.er(e);
            return '';
        }
        if (date.isAfter(todayMidnight)) {
            return this.translateService.instant('date_service.today');
        }
        if (date.isAfter(yesterdayMidnight)) {
            return this.translateService.instant('date_service.yesterday');
        }
        return date.format(args);
    }
}
