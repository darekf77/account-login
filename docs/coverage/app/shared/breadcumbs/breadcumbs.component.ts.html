    private getDisplayName(instruction: any) {
        if (BreadcrumbsComponent.staticCrumb) {
            let routeName = instruction.component.routeName;
            if (BreadcrumbsComponent.staticCrumb.hasOwnProperty(routeName)) {
                for (let param in BreadcrumbsComponent.staticCrumb[routeName].params) {
                    if (instruction.component.params.hasOwnProperty(param)
                        && instruction.component.params[param].toString()
                        === BreadcrumbsComponent.staticCrumb[routeName].params[param].toString()) {
                        return BreadcrumbsComponent.staticCrumb[routeName].displayName;
                    }
                }
            }
        }
        return instruction.component.routeData.get('displayName');
    }

    private _getLinkParams(instructions, until): string[] {
        let linkParams = [];
        instructions.forEach((item, index) => {
            let component = item.component;
            let routePrefix: string;
 
            if (component.routeData.get('parent') && index < until) {
                return;
            }
 
            if (index <= until) {
                routePrefix = (index === 0) ? '/' : '';
                linkParams.push(routePrefix + component.routeName);
 
                if (!this._isEmpty(component.params)) {
                    // linkParams.push(component.params);
                    let tmpParams: any = {};
                    let skipParam: boolean;
                    for (let param of item.component.params) {
                        if (item.component.params.hasOwnProperty(param)) {
                            skipParam = false;
                            for (let urlParam of item.urlParams) {
                                if (urlParam.indexOf(param) === 0) {
                                    skipParam = true;
                                    break;
                                }
                            }
                            if (!skipParam) {
                                tmpParams[param] = item.component.params[param];
                            }
                        }
 
                    }
 
                    if (tmpParams) {
                        linkParams.push(tmpParams);
                    }
                    // console.log('tmp params: ----------------');
                    // console.log(tmpParams);
                }
                // console.log('---- link params ========');
                // console.log(linkParams);
            }
        });
 
        return linkParams;
    }
 
    private _isEmpty(obj) {
        for (let prop in obj) {
            if (obj.hasOwnProperty(prop))
                return false;
        }
 
        return true;
    }
