import { Resource } from 'ng2-rest/ng2-rest';
import { Worker } from './worker';
import { ENDPOINTS } from '../../app.component';
import { mockController } from './mock/eniro-workers-mock';
 
const modelName = 'eniro/workers';
 
 
@Injectable()
export class WorksersService {
 
    model = {
        getWorkerBy: (id) => this.rest.api(ENDPOINTS.API, modelName,
            'Get eniro workers, reprezentants by id ')
            .mock(require('./mock/eniro-workers.mock.json'), 300, mockController)
            .get(id)
    };
 
    constructor(private rest: Resource<ENDPOINTS, Worker, Worker[]>) {
        this.rest.add(ENDPOINTS.API, modelName, 'INTERNAL ENIRO');
    }
}
