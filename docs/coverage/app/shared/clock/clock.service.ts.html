import { Injectable } from '@angular/core';
import { TimeFormat } from './time-format.enum';
 
 
@Injectable()
export class ClockService {
 
    constructor() { }
 
    getTimeInFormat(date: Date, format: TimeFormat = TimeFormat.HH_MM) {
        if (!date) throw Error('Date should be defined');
        let d = date.getDate();
        let day = date.getDay();
        let h: any = date.getHours();
        if (h < 10) {
            h = '0' + h;
        }
        let m: any = date.getMinutes();
        if (m < 10) {
            m = '0' + m;
        }
        let s: any = date.getSeconds();
        if (s < 10) {
            s = '0' + s;
        }
        if (format === TimeFormat.HH_MM) return h + ':' + m;
        if (format === TimeFormat.HH_MM_SS) return h + ':' + m + ':' + s;
    }
 
 
}
