import { Validators, FormControl, FormGroup, FormBuilder } from '@angular/forms';
 
import { FormInputControl } from './form-input-control.interface';
import { FormInputConfig } from './form-input-config.interface';
 
export abstract class FormFactoryAbstract {
    protected inputControl: FormInputControl = {};
    protected inputConfig: FormInputConfig[] = [];
 
    constructor(protected fb: FormBuilder) {
    }
 
    create(data: any): FormGroup {
        this.dataToInputConfig(data);
        this.createFormControlByConfig(this.inputConfig);
        return this.getFormGroup();
    }
 
    // @override: może zaistnieć potrzeba nadpisania poniższej metody jeżeli "data" będzie
    // "zagnieżdzonym" obiektem !!!
    protected dataToInputConfig(data: any): void {
        for (let i in this.inputConfig) {
            let item: FormInputConfig = this.inputConfig[i];
            if (data[item.name]) {
                item.value = data[item.name];
            }
        }
    }
 
    protected createFormControlByConfig(inputsConfig: FormInputConfig[]): void {
        for (let i in inputsConfig) {
            let inputConfig: FormInputConfig = inputsConfig[i];
            this.createFormControl(inputConfig.name, inputConfig.value, inputConfig.validators);
        }
    }
 
    protected createFormControl(name: string, value: any, validators: any[]): void {
        this.inputControl[name] = new FormControl(
            value,
            Validators.compose(validators)
        );
    }
 
    protected getFormGroup(): FormGroup {
        return this.fb.group(this.inputControl);
    }
}
