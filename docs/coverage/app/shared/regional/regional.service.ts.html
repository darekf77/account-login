import { Injectable, Inject } from '@angular/core';
import { Observable } from 'rxjs';
 
import { Resource } from 'ng2-rest/ng2-rest';
 
import { ENDPOINTS } from '../../app.component';
import { RegionalData } from './regional-data';
import { getByPostcode } from './mock';
import { Category } from './category';
 
const modelName = '/regional/data';
 
 
interface RegionaServiceInterface {
    countries: () => Observable<RegionalData[]>;
    voivodeships: (countryId: number) => Observable<RegionalData[]>;
    voivodeship: (postOfficeId: number) => Observable<RegionalData>;
    counties: (voivodeshipId: number) => Observable<RegionalData[]>;
    countiesByPostCode: (postcode: string) => Observable<RegionalData[]>;
    communies: (countyId: number) => Observable<RegionalData[]>;
    communiesByPostCode: (postcode: string) => Observable<RegionalData[]>;
    cities: (city: string, postalCode?: string) => Observable<RegionalData[]>;
    postOffices: (postOffice: string, postalCode?: string) => Observable<RegionalData[]>;
    communiesByCityId: (cityId: number) => Observable<RegionalData[]>;
    addressTypes: () => Observable<RegionalData[]>;
    all: (postcode: string, countryId: number) => Observable<RegionalData[]>;
}
 
@Injectable()
export class RegionalService {
 
    model: RegionaServiceInterface;
 
    constructor(private rest: Resource<ENDPOINTS, RegionalData, RegionalData[]>) {
        rest.add(ENDPOINTS.API, modelName, 'REGIONAL DATA');
        this.model = <RegionaServiceInterface>{
            countries: () => this.rest
                .api(ENDPOINTS.API, modelName, 'Get all countries')
                .mock(require('./mock/countries.mock.json'), 500)
                .query([{ caregory: Category.Country }]),
 
            voivodeships: (countryId: number) => this.rest
                .api(ENDPOINTS.API, modelName, 'Get all voivodeships by id of country')
                .mock(require('./mock/voivodeships.mock.json'), 500)
                .query([{ caregory: Category.Voivodeship, countryId }]),
 
            voivodeship: (postOfficeId: number) => this.rest
                .api(ENDPOINTS.API, modelName, 'Get all voivodeships by id of postoffice')
                .mock(require('./mock/voivodeship.mock.json'), 500)
                .get([{ caregory: Category.Voivodeship, postOfficeId }]),
 
            counties: (voivodeshipId: number) => this.rest
                .api(ENDPOINTS.API, modelName, 'Get counties by voivodeship id')
                .mock(require('./mock/counties.mock.json'), 500)
                .query([{ caregory: Category.Commune, voivodeshipId }]),
 
            countiesByPostCode: (postcode: string) => this.rest
                .api(ENDPOINTS.API, modelName, 'Get counties by postcode')
                .mock(require('./mock/counties.mock.json'), 500)
                .query([{ caregory: Category.Commune, postcode }]),
 
            cities: (city: string, postalCode?: string) => this.rest
                .api(ENDPOINTS.API, modelName, 'Get cities by post code')
                .mock(require('./mock/cities-data.mock.json'), 500)
                .query([{ caregory: Category.City, city, postalCode }]),
 
            postOffices: (postOffice: string, postalCode?: string) => this.rest
                .api(ENDPOINTS.API, modelName, 'Get post-offices by post code')
                .mock(require('./mock/post-offices-data.mock.json'), 500)
                .query([{ caregory: Category.City, postOffice, postalCode }]),
 
            communies: (countyId: number) => this.rest
                .api(ENDPOINTS.API, modelName, 'Get communies by county id')
                .mock(require('./mock/communies.mock.json'), 500)
                .query([{ caregory: Category.Commune, countyId }]),
 
            communiesByPostCode: (postcode: string) => this.rest
                .api(ENDPOINTS.API, modelName, 'Get communies by post code')
                .mock(require('./mock/communies.mock.json'), 500)
                .query([{ caregory: Category.Commune, postcode }]),
 
            communiesByCityId: (cityId: number) => this.rest
                .api(ENDPOINTS.API, modelName, 'Get communies by city id')
                .mock(require('./mock/communies.mock.json'), 500)
                .query([{ caregory: Category.Commune, cityId }]),
 
            addressTypes: () => this.rest
                .api(ENDPOINTS.API, modelName, 'Get all addres types')
                .mock(require('./mock/address-types.mock.json'), 500)
                .query([{ caregory: Category.AddressType }]),
 
            all: (postcode: string, countryId: number) =>
                this.rest.api(ENDPOINTS.API, modelName, 'Get all of WHAT!!!! ?????? ')
                    .mock(require('./mock/regional-data.mock.json'), 500, getByPostcode)
                    .query([{ postcode, countryId }])
        };
    }
 
 
 
 
 
}
