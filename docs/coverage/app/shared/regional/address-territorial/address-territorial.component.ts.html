import { Component, Input, OnDestroy, OnInit } from '@angular/core';
import { FormControl } from '@angular/forms';
import { LoadingInfoComponent } from 'ng2-encrm-components';
 
import { RegionalService } from './../regional.service';
import { RegionalData } from './../regional-data';
import { InputSelectOption } from '../../input/select/input-select-option.interface';
 
// const log = Log.create('register data cmp');
 
@Component({
    selector: 'ecrm-address-territorial',
    templateUrl: './address-territorial.component.html'
})
 
export class AddressTerritorialComponent implements OnInit, OnDestroy {
 
    @Input() formControlCountry: FormControl;
    @Input() formControlVoivodeship: FormControl;
    @Input() formControlCounty: FormControl;
    @Input() formControlCommune: FormControl;
 
    @Input() counties: InputSelectOption[] = [];
    @Input() communies: InputSelectOption[] = [];
    protected countries: InputSelectOption[] = [];
    protected voivodeships: InputSelectOption[] = [];
 
    protected countyLabel = 'count';
    protected communeLabel = 'commune';
    protected voivodeshipLabel = 'voivodeship';
    protected countryLabel = 'country';
 
    protected subCountry: any;
    protected subVoivodeship: any;
    protected subCounties: any;
    protected subCommunies: any;
 
    constructor(private regionalService: RegionalService) {
 
    }
 
    ngOnInit() {
 
        this.subscribeVoivodeshipData();
        this.subscribeCountryData();
    }
 
    ngOnDestroy() {
        if (this.subCounties) this.subCounties.unsubscribe();
        if (this.subCommunies) this.subCommunies.unsubscribe();
        if (this.subVoivodeship) this.subVoivodeship.unsubscribe();
        if (this.subCountry) this.subCountry.unsubscribe();
    }
 
    protected subscribeVoivodeshipData() {
        if (!this.formControlCounty.value) {
            return;
        }
 
        LoadingInfoComponent.startLoading(this.voivodeshipLabel);
        this.subVoivodeship = this.regionalService.model.voivodeships(this.formControlCounty.value).subscribe((voivodeships: RegionalData[]) => {
            this.regionalDataToInputSelectOptionFormat('voivodeships', voivodeships);
            LoadingInfoComponent.stopLoading(this.voivodeshipLabel);
        });
    }
 
    protected subscribeCountryData(): void {
        LoadingInfoComponent.startLoading(this.voivodeshipLabel);
        this.subCountry = this.regionalService.model.countries().subscribe((countries: RegionalData[]) => {
            this.regionalDataToInputSelectOptionFormat('countries', countries);
            LoadingInfoComponent.stopLoading(this.voivodeshipLabel);
        });
    }
 
    protected handlerCountryCodeChanged(countryId: number): void {
        this.subscribeVoivodeshipData();
    }
 
    protected getKeyLabel(key: string): string {
        return 'content.client.card.accordions.' + key;
    }
 
    protected regionalDataToInputSelectOptionFormat(key: string, data: RegionalData[]) {
        let collection = [];
        for (let item of data) {
            collection.push({id: item.id, value: item.name});
        }
 
        switch (key) {
            case 'counties':
                this.counties = collection;
                break;
            case 'communies':
                this.communies = collection;
                break;
            case 'voivodeships':
                this.voivodeships = collection;
                break;
            case 'countries':
                this.countries = collection;
                break;
            default:
                break;
        }
    }
}
