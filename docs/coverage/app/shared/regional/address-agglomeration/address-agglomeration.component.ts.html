import { Component, OnInit, Input } from '@angular/core';
import { RegionalService } from '../regional.service';
import { FormControl } from '@angular/forms';
import { Observable } from 'rxjs';
 
import { Category } from '../category';
 
@Component({
    selector: 'ecrm-address-agglomeration',
    templateUrl: './address-agglomeration.component.html',
    styles: [require('./address-agglomeration.component.scss')],
})
 
export class AddressAgglomerationComponent implements OnInit {
    @Input() postalCode: FormControl;
    @Input() city: FormControl;
    @Input() cityId: FormControl;
    @Input() postOfficeId: FormControl;
    @Input() postOffice: FormControl;
    @Input() isPostalCodeRequired: boolean = false;
    @Input() isCityRequired: boolean = false;
    @Input() isPostOfficeRequired: boolean = false;
    @Input() countryId: FormControl;
 
    protected typeaheadOptionsLimit = 10;
 
    protected statesCityComplex: Array<any> = []; // kolekcja wartosci wyswietlonych jako lista podpowiedzi
    protected dataCitySource: Observable<any>;
    protected asyncCitySelected: string = ''; // fraza po ktorej nastepuje wyszukiwanie
    protected typeaheadCityLoading: boolean = false;
 
    protected statesPostOfficeComplex: Array<any> = []; // kolekcja wartosci wyswietlonych jako lista podpowiedzi
    protected dataPostOfficeSource: Observable<any>;
    protected asyncPostOfficeSelected: string = ''; // fraza po ktorej nastepuje wyszukiwanie
    protected typeaheadPostOfficeLoading: boolean = false;
 
    constructor(private regionalService: RegionalService) {
 
    }
 
    ngOnInit() {
        this.initValues();
        this.initSource();
        this.subscribePostalCodeChanges();
    }
 
    manuallyEnteredDataCity(e) {
        this.city.setValue(e);
    }
 
    changeTypeaheadCityLoading(e) {
        this.typeaheadCityLoading = e;
    }
 
    changeTypeaheadCityNoResults(e) {
        this.cityId.setValue(0);
    }
 
    typeaheadCityOnSelect(e: {item: {id: any, name: any}}) {
        this.cityId.setValue(e.item.id);
        this.city.setValue(e.item.name);
    }
 
    manuallyEnteredDataPostOffice(e) {
        this.postOffice.setValue(e);
    }
 
    changeTypeaheadPostOfficeLoading(e) {
        this.typeaheadPostOfficeLoading = e;
    }
 
    changeTypeaheadPostOfficeNoResults(e) {
        this.postOfficeId.setValue(0);
    }
 
    typeaheadPostOfficeOnSelect(e: {item: {id: any, name: any}}) {
        this.postOfficeId.setValue(e.item.id);
        this.postOffice.setValue(e.item.name);
    }
 
    protected initValues() {
        this.asyncCitySelected = this.city.value.toString();
        this.asyncPostOfficeSelected = this.postOffice.value.toString();
    }
 
    protected initSource(): void {
        this.dataCitySource = Observable.create((observer: any) => {
            this.loadByCity(observer, this.asyncCitySelected);
        }).mergeMap((token: string) => this.getStatesCityAsObservable(token));
 
        this.dataPostOfficeSource = Observable.create((observer: any) => {
            this.loadByPostOffice(observer, this.asyncPostOfficeSelected);
        }).mergeMap((token: string) => this.getStatesPostOfficeAsObservable(token));
    }
 
    protected subscribePostalCodeChanges() {
        this.postalCode.valueChanges.subscribe((res) => {
            if (!this.postalCode.value) {
                this.asyncCitySelected = '';
                this.asyncPostOfficeSelected = '';
            } else if (this.postalCode.valid) {
                this.changeTypeaheadCityLoading(true);
                this.changeTypeaheadPostOfficeLoading(true);
 
                this.regionalService.model.all(this.postalCode.value, this.countryId.value).subscribe(results => {
 
                    if (results.length === 0) {
                        this.setCity(0, '');
                        this.setPostalCode(0, '');
                    } else {
                        let isCity = false;
                        let isPostOffice = false;
                        results.forEach(res => {
                            if (res.category === Category.PostOffice && !isPostOffice) {
                                this.setPostalCode(res.id, res.name);
                                isPostOffice = true;
                            } else if (res.category === Category.City && !isCity) {
                                this.setCity(res.id, res.name);
                                isCity = true;
                            }
                        });
                    }
 
                    this.changeTypeaheadCityLoading(false);
                    this.changeTypeaheadPostOfficeLoading(false);
                });
            }
        });
    }
 
    protected loadByCity(observer: any, token: any): void {
        this.regionalService.model.cities(token, this.postalCode.value).subscribe((res: any) => {
            this.statesCityComplex = res;
            observer.next(token);
        });
    }
 
    protected getStatesCityAsObservable(token: string): Observable<any> {
        return Observable.of(
            this.statesCityComplex.filter((state: any) => {
                let query = new RegExp(token, 'ig');
                return query.test(state.name);
            })
        );
    }
 
    protected loadByPostOffice(observer: any, token: any): void {
        this.regionalService.model.postOffices(token, this.postalCode.value).subscribe((res: any) => {
            this.statesPostOfficeComplex = res;
            observer.next(token);
        });
    }
 
    protected getStatesPostOfficeAsObservable(token: string): Observable<any> {
        return Observable.of(
            this.statesPostOfficeComplex.filter((state: any) => {
                let query = new RegExp(token, 'ig');
                return query.test(state.name);
            })
        );
    }
 
    protected setCity(id: number, value: string): void {
        this.cityId.setValue(id);
        this.city.setValue(value);
        this.asyncCitySelected = value;
    }
 
    protected setPostalCode(id: number, value: string): void {
        this.postOfficeId.setValue(id);
        this.postOffice.setValue(value);
        this.asyncPostOfficeSelected = value;
    }
 
    protected isRequiredSign(key: string) {
        let isRequired: boolean = false;
        switch (key) {
            case 'postalCode':
                isRequired = this.isPostalCodeRequired;
                break;
            case 'city':
                isRequired = this.isCityRequired;
                break;
            case 'postOffice':
                isRequired = this.isPostOfficeRequired;
                break;
            default:
                break;
        }
 
        return (isRequired) ? '*' : '';
    }
}
