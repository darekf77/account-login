import { Component, OnInit, Input, Output, OnDestroy, EventEmitter, OnChanges } from '@angular/core';
import { FormGroup } from '@angular/forms';
 
import { Subscription, Subject } from 'rxjs';
 
import { Log, Level } from 'ng2-logger/ng2-logger';
import { LoadingInfoComponent } from 'ng2-encrm-components';
import { debounceable } from '../../../shared/utils';
import { RegionalData } from '../regional-data';
import { RegionalService } from '../regional.service';
import { Category } from '../category';
 
const log = Log.create('post-code widget', Level.__NOTHING);
 
const id: string = 'post-code-input';
 
@Component({
    selector: 'post-code-input',
    template: require(`./post-code-input.component.html`),
    styles: [require(`./post-code-input.component.scss`)]
})
export class PostCodeInputComponent implements OnInit, OnDestroy, OnChanges {
 
    @Output() postCodeChanged: EventEmitter<string> = new EventEmitter<string>();
 
    @Output() cityIdChanged: EventEmitter<number> = new EventEmitter<number>();
    @Output() postOfficeIdChanged: EventEmitter<number> = new EventEmitter<number>();
 
    @Input() componentId: string;
    @Input() postcode: string;
    @Input() countryId: number;
 
    @Input() cityId: number;
    @Input() postOfficeId: number;
 
    @Input() cities: RegionalData[];
    @Input() postOffices: RegionalData[];
 
    private static externalChange: Subject<string> = new Subject<string>();
    private static externalChangeObs = PostCodeInputComponent.externalChange.asObservable();
 
    citySelectBoxChanged(e) {
        this.cityIdChanged.emit(e);
    }
 
    postofficeSelectBoxChanged(e) {
        this.postOfficeIdChanged.emit(e);
    }
 
    ngOnChanges(newValue) {
        log.d('ngOnChanges', newValue);
    }
 
    constructor(private regio: RegionalService) {
        PostCodeInputComponent
            .externalChangeObs.subscribe(() => {
                setTimeout(() => {
                    log.d('hello static observable', this.postcode)
                    if (this.postcode && this.postcode.trim() !== '' && this.postcode !== this.currentSearch) {
                        this.fastSearch(this.postcode);
                    }
                });
 
            });
    }
 
    handlers: Subscription[] = [];
 
    ngOnInit() {
 
    }
 
    ngOnDestroy() {
 
    }
 
    t(key: string) {
        return `${key}${this.componentId}`;
    }
 
    private prevCities: RegionalData[] = [];
    private prevPostOfficesOffices: RegionalData[] = [];
 
    private currentSearch: string;
 
    @debounceable(500, undefined)
    searchRegionalData(e: string) {
        this.currentSearch = e;
        PostCodeInputComponent.externalChange.next(this.currentSearch);
        this.postCodeChanged.emit(e);
        this.fastSearch(e);
    }
 
    fastSearch(e) {
        if (e.trim() === '') {
            this.postOffices.length = 0;
            this.cities.length = 0;
            this.cityIdChanged.emit(this.cityId);
            this.postOfficeIdChanged.emit(this.postOfficeId);
        } else {
            log.d('searchRegionalData after debouce', e);
            LoadingInfoComponent.startLoading(this.t('city'));
            LoadingInfoComponent.startLoading(this.t('postoffice'));
            this.regio.model.all(e, this.countryId).subscribe(results => {
                this.prevCities = this.cities || [];
                this.prevPostOfficesOffices = this.postOffices || [];
                this.postOffices = [];
                this.cities = [];
                results.forEach(res => {
                    if (res.category === Category.PostOffice) this.postOffices.push(res);
                    if (res.category === Category.City) this.cities.push(res);
                });
                if (!checkEquality(this.postOffices, this.prevPostOfficesOffices)) {
                    this.postOfficeId = this.postOffices.length > 0 ? this.postOffices[0].id : undefined;
                    this.postOfficeIdChanged.emit(this.postOfficeId);
                }
 
                if (!checkEquality(this.cities, this.prevCities)) {
                    this.cityId = this.cities.length > 0 ? this.cities[0].id : undefined;
                    this.cityIdChanged.emit(this.cityId);
                }
 
                LoadingInfoComponent.stopLoading(this.t('city'));
                LoadingInfoComponent.stopLoading(this.t('postoffice'));
            });
        }
    }
 
 
    comunies: RegionalData[] = [];
    counties: RegionalData[] = [];
    voivodeships: RegionalData[] = [];
 
 
    get = {
 
        CommunitiesBy: (countyId, after: () => void = undefined) => {
            log.i('communites changed');
            let communiesO = this.regio.model.communies(countyId);
            let communiesH = communiesO.subscribe(communies => {
                this.comunies = communies;
                setTimeout(() => communiesH.unsubscribe());
                if (after !== undefined) after();
            });
            LoadingInfoComponent.setOn(this.t('commune'), communiesO);
            if (this.handlers.indexOf(communiesH) === -1) this.handlers.push(communiesH);
        },
 
        VoivodeshipsByCountry: (countryId, after: () => void = undefined) => {
            log.i('voivodeship changed');
            let voivodeshipO = this.regio.model.voivodeships(countryId);
            let voivodeshipsH = voivodeshipO.subscribe(
                voivodeships => {
                    this.voivodeships = voivodeships;
                    if (after !== undefined) after();
                    setTimeout(() => voivodeshipsH.unsubscribe());
                });
            LoadingInfoComponent.setOn(this.t('voivodeship'), voivodeshipO);
            if (this.handlers.indexOf(voivodeshipsH) === -1) this.handlers.push(voivodeshipsH);
        },
 
        CountiesBy: (voivodeshipId, after: () => void = undefined) => {
            log.i('counties changed');
            let countiesO = this.regio.model.counties(voivodeshipId);
            let countiesH = countiesO.subscribe(counties => {
                this.counties = counties;
                setTimeout(() => countiesH.unsubscribe());
                if (after !== undefined) after();
            });
            LoadingInfoComponent.setOn(this.t('county'), countiesO);
            if (this.handlers.indexOf(countiesH) === -1) this.handlers.push(countiesH);
        }
    };
 
}
 
 
function checkEquality(a: RegionalData[], b: RegionalData[]) {
    let elemNotFound = false;
    a.forEach(e => {
        if (!b.includes(e)) {
            elemNotFound = true;
            return;
        }
    });
    if (elemNotFound) return false;
    b.forEach(e => {
        if (!a.includes(e)) {
            elemNotFound = true;
            return;
        }
    });
    if (elemNotFound) return false;
    return true;
}
