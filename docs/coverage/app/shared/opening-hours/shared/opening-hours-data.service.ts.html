import { Injectable } from '@angular/core';
 
import { Resource } from 'ng2-rest/ng2-rest';
 
import { ENDPOINTS } from './../../../app.component';
import { OpeningHours } from './opening-hours.interface';
 
const modelName: string = '/clientcard/openingHoursData';
 
@Injectable()
export class OpeningHoursDataService {
 
    model = {
        getByElid: (elid: number) => this.rest
            .api(ENDPOINTS.API, modelName, 'Get opening hours by elid')
            .mock(require('./mock/opening-hours-data.mock.json'), 200)
            .get([{ elid }]),
 
        update: (elid: number, model: OpeningHours) => this.rest
            .api(ENDPOINTS.API, modelName, 'Update opening hours')
            .mock(require('./mock/opening-hours-data.mock.json'), 200)
            .update(model, [{ elid }])
 
    };
 
    constructor(private rest: Resource<ENDPOINTS, OpeningHours, {}>) {
        rest.add(ENDPOINTS.API, modelName, 'Opening Hours Data', undefined, '');
    }
}
