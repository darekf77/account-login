import { Component, Input, OnInit, EventEmitter, Output, OnDestroy } from '@angular/core';
import { Subscription } from 'rxjs';
 
import { OpeningHours } from './../shared/opening-hours.interface';
import { OpeningHoursDataService } from './../shared/opening-hours-data.service';
 
@Component({
    selector: 'ecrm-opening-hours-edit',
    templateUrl: './edit.component.html',
    styleUrls: ['./edit.component.scss']
})
 
export class OpeningHoursEditComponent implements OnInit, OnDestroy {
    @Input() elid: number;
    @Input() openingHours: OpeningHours;
    @Output() isSaved: EventEmitter<any> = new EventEmitter<any>();
 
    protected days: string[] = ['monday', 'tuesday', 'wednesday', 'thursday', 'friday', 'saturday', 'sunday'];
    protected proposalOpeningHoursFrom: string = '00:00';
    protected proposalOpeningHoursTo: string = '00:00';
    protected proposalOpeningHoursBreakFrom: string = '00:00';
    protected proposalOpeningHoursBreakTo: string = '00:00';
    protected frequency: number = 0;
    protected sub: Subscription[] = [];
 
    constructor(protected openingHoursService: OpeningHoursDataService) {
 
    }
 
    ngOnInit() {
        if (this.openingHours
            && this.openingHours.frequency.daily === false
            && this.openingHours.frequency.working_days === false) {
            this.setFrequency(true, false);
        }
 
        this.setDefaultFrequency();
    }
 
    ngOnDestroy() {
        this.sub.forEach((s) => s.unsubscribe());
        this.sub = [];
    }
 
    handlerProposalHoursFrom(val: string): void {
        this.changeAllHours('from', val);
    }
 
    handlerProposalHoursTo(val: string): void {
        this.changeAllHours('to', val);
    }
 
    handlerProposalBreakHoursFrom(val: string): void {
        this.changeAllHours('from', val, true);
    }
 
    handlerProposalBreakHoursTo(val: string): void {
        this.changeAllHours('to', val, true);
    }
 
    handlerFrequency(): void {
        switch (this.frequency) {
            case 1:
                this.setFrequency(true, false);
                break;
            case 2:
                this.setFrequency(false, true);
                break;
            default:
                break;
        }
    }
 
    handlerChangeSeasonDate(key: string, value: any): void {
        switch (key) {
            case 'beginning':
                this.openingHours.season.beginning = new Date(value).getTime();
                break;
            case 'end':
                this.openingHours.season.end = new Date(value).getTime();
                break;
            default:
                break;
        }
    }
 
    handlerSave(): void {
        this.sub.push(this.openingHoursService.model.update(this.elid, this.openingHours).subscribe((res) => {
            this.isSaved.emit({});
        }));
    }
 
    protected validOpeningHours(value: string) {
        let EMAIL_REGEXP = /^[0-9]{1,2}:[0-9]{2,2}$/i;
        return EMAIL_REGEXP.test(value);
    }
 
    protected changeAllHours(key: string, value: string, isBreak: boolean = false): void {
        if (this.validOpeningHours(value)) {
            for (let i in this.days) {
                let day: string = this.days[i];
 
                if (!isBreak) {
                    this.openingHours.days[day][key] = value;
                } else {
                    this.openingHours.days[day].break[key] = value;
                }
            }
        }
    }
 
    protected setDefaultFrequency(): void {
        if (this.openingHours.frequency.daily) {
            this.frequency = 1;
        }
        else Eif (this.openingHours.frequency.working_days) {
            this.frequency = 2;
        }
    }
 
    protected setFrequency(daily: boolean, workingDays: boolean): void {
        this.openingHours.frequency.daily = daily;
        this.openingHours.frequency.working_days = workingDays;
    }
}
