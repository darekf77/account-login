import { Injectable } from '@angular/core';
 
import { TranslateService } from 'ng2-translate/ng2-translate';
import { ColumnDefinition } from 'ng2-encrm-components/components/edit-table/edit-table.protocols';
import { times, isString } from 'lodash';
 
 
 
/**
 * A service for project-wide, component-independent helper methods.
 *
 */
@Injectable()
export class EcrmHelpers {
 
    constructor(private translateService: TranslateService) {}
 
 
    translateEditTableColumnDefinitions(...listOfColumnDefinitions: ColumnDefinition[][]): void {
        listOfColumnDefinitions.forEach(columnDefinitions => {
            columnDefinitions.forEach(col => {
                if (col.header) {
                    this.translateService.get(col.header).subscribe(translation => {
                        col.header = translation;
                    });
                }
 
                if (col.invalidDataMsg) {
                    this.translateService.get(col.invalidDataMsg).subscribe(translation => {
                        col.invalidDataMsg = translation;
                    });
                }
            });
        });
    }
 
 
 
    parsePesel(pesel: string) {
        let parsed = {
            birthYear: undefined as number,
            birthMonth: undefined as number,
            birthDay: undefined as number,
            isMan: undefined as boolean,
            isFormatValid: undefined as boolean,
            isControlSumValid: undefined as boolean,
            isPeselValid: undefined as boolean
        };
 
 
        // validation
 
        parsed.isFormatValid = isString(pesel) && /^[0-9]{11}$/.test(pesel);
 
        if (!parsed.isFormatValid) {
            parsed.isPeselValid = false;
            return parsed;
        }
 
 
        let controlSumFactors = [9, 7, 3, 1, 9, 7, 3, 1, 9, 7];
 
        let calculatedControlSum = 0;
        times(10, i => calculatedControlSum += parseInt(pesel[i], 10) * controlSumFactors[i]);
        calculatedControlSum = calculatedControlSum % 10;
 
        parsed.isControlSumValid = calculatedControlSum === parseInt(pesel[10], 10);
 
        if (!parsed.isControlSumValid) {
            parsed.isPeselValid = false;
            return parsed;
        }
 
 
        parsed.isPeselValid = true;
 
 
        // data extraction
 
        let birthCentury;
 
        switch (parseInt(pesel[2], 10)) {
            case 8:
            case 9:
                birthCentury = 1800;
                break;
            case 0:
            case 1:
                birthCentury = 1900;
                break;
            case 2:
            case 3:
                birthCentury = 2000;
                break;
            default:
        }
 
        parsed.birthYear = birthCentury + parseInt(pesel.slice(0, 2), 10);
        parsed.birthMonth = parseInt(pesel.slice(2, 4), 10) % 20;
        parsed.birthDay = parseInt(pesel.slice(4, 6), 10);
 
        parsed.isMan = parseInt(pesel[9], 10) % 2 === 1;
 
        return parsed;
    }
 
}
