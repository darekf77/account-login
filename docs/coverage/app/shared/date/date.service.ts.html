import { Injectable } from '@angular/core';
import { DateFormat } from './date-format.enum';
import { TranslateService } from 'ng2-translate/ng2-translate';
 
@Injectable()
export class DateService {
 
    constructor(private translate: TranslateService) {
    }
 
    getDayOfWeek(n: Number) {
        if (n === 0) return this.translate.instant('days_of_week.sunday');
        if (n === 1) return this.translate.instant('days_of_week.monday');
        if (n === 2) return this.translate.instant('days_of_week.thuesday');
        if (n === 3) return this.translate.instant('days_of_week.wednesday');
        if (n === 4) return this.translate.instant('days_of_week.thursday');
        if (n === 5) return this.translate.instant('days_of_week.friday');
        if (n === 6) return this.translate.instant('days_of_week.saturday');
    }
 
    getMonthOfYear(n: number) {
        if (n === 0) return this.translate.instant('months_of_year.january');
        if (n === 1) return this.translate.instant('months_of_year.february');
        if (n === 2) return this.translate.instant('months_of_year.march');
        if (n === 3) return this.translate.instant('months_of_year.april');
        if (n === 4) return this.translate.instant('months_of_year.may');
        if (n === 5) return this.translate.instant('months_of_year.june');
        if (n === 6) return this.translate.instant('months_of_year.july');
        if (n === 7) return this.translate.instant('months_of_year.august');
        if (n === 8) return this.translate.instant('months_of_year.september');
        if (n === 9) return this.translate.instant('months_of_year.october');
        if (n === 10) return this.translate.instant('months_of_year.november');
        if (n === 11) return this.translate.instant('months_of_year.december');
    }
 
    getWeekOfYear(date) {
        let target = new Date(date.valueOf()),
            dayNumber = (date.getUTCDay() + 6) % 7,
            firstThursday;
 
        target.setUTCDate(target.getUTCDate() - dayNumber + 3);
        firstThursday = target.valueOf();
        target.setUTCMonth(0, 1);
 
        if (target.getUTCDay() !== 4) {
            target.setUTCMonth(0, 1 + ((4 - target.getUTCDay()) + 7) % 7);
        }
 
        return Math.ceil((firstThursday - target.getTime()) /
                (7 * 24 * 3600 * 1000)) + 1;
    }
 
    getDayOfYear(date) {
        let now = new Date();
        let start = new Date(now.getFullYear(), 0, 0);
        let diff = now.getTime() - start.getTime();
        let oneDay = 1000 * 60 * 60 * 24;
        return Math.floor(diff / oneDay);
    }
 
    getDateInFormat(date: Date, format: DateFormat = DateFormat.WEEKDAY_DAY__MONTH_YEAR) {
        if (!date) throw Error('Date should be defined');
        if (format === DateFormat.WEEKDAY_DAY__MONTH_YEAR) {
            let year = date.getFullYear();
            let month = date.getMonth();
            let monthOfYear = this.getMonthOfYear(month);
            let day = date.getDate();
            let dayW = date.getDay();
            let dayOfWeek = this.getDayOfWeek(dayW);
            return `${dayOfWeek} ${day} ${monthOfYear} ${year}`;
        }
        if (format === DateFormat.YEARDAY_YEARWEEK) {
            let day = this.getDayOfYear(date);
            let week = this.getWeekOfYear(date);
            let dayOfYearL = this.translate.instant('date_service.day_of_year');
            let weekL = this.translate.instant('date_service.week');
            return `${day} ${dayOfYearL}, ${week} ${weekL}`;
        }
 
    }
 
 
}
