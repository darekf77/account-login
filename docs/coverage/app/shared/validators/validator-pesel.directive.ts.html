import { Directive, forwardRef } from '@angular/core';
 
import { NG_VALIDATORS, FormControl } from '@angular/forms';
 
import { EcrmHelpers } from '../../shared/utils';
 
@Directive({
  selector: '[validatePesel][ngModel], [validatePesel][formControl]',
  providers: [
    { provide: NG_VALIDATORS, useExisting: forwardRef(() => PeselValidatorDirective), multi: true }
  ]
})
export class PeselValidatorDirective {
 
  validator: Function;
 
  constructor(private ecrmHelpers: EcrmHelpers) {
    this.validator = (pesel: FormControl) => {
      if (!pesel.value || ecrmHelpers.parsePesel(pesel.value).isPeselValid) {
        return null;
      } else {
        return {incorrectPesel: 'incorrectPesel'}
      }
    };
  }
 
  validate(c: FormControl) {
    return this.validator(c);
  }
}
