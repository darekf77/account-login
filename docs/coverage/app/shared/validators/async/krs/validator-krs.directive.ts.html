import { Directive, forwardRef } from '@angular/core';
import { ENDPOINTS } from '../../../../app.component';
import { Resource, MockRequest, MockResponse } from 'ng2-rest/ng2-rest';
 
import { NG_ASYNC_VALIDATORS, FormControl } from '@angular/forms';
 
const modelName = 'checkKrs';
 
function mockController(req: MockRequest<any>): MockResponse {  // data: any, params: any
    let krs = req.params ? req.params['id'] : undefined;
    let d = (krs === '111');
    console.log(`is valid krs:${krs} : `, d);
    return { data: d };
}
 
 
function validateEmailFactory(rest: Resource<ENDPOINTS, boolean, {}>) {
    return (c: FormControl) => {
 
 
        return new Promise((resolve, reject) => {
 
            let krs = (!c.value || c.value === null) ? '' : c.value.toString() || '';
 
            if (krs !== '') {
                rest.api(ENDPOINTS.API, modelName, 'Validate krs number async from db')
                    .mock('{}', 200, mockController)
                    .get(c.value).subscribe(
                    res => {
                        console.log('resovled!', res);
                        (res) ? resolve(null) : resolve({ invalidKrs: true });
                    },
                    err => {
                        console.log('rejection ');
                        resolve({ somethingbad: true });
                    });
            } else resolve(null);
 
 
        });
 
    };
}
 
@Directive({
    selector: '[validateKrsAsync][ngModel],[validateKrsAsync][formControl]',
    providers: [
        {
            provide: NG_ASYNC_VALIDATORS,
            useExisting: forwardRef(() => KrsAsyncValidator),
            multi: true
        }
    ]
})
export class KrsAsyncValidator {
 
    validator: Function;
 
    constructor(rest: Resource<ENDPOINTS, boolean, {}>) {
        rest.add(ENDPOINTS.API, modelName, 'VALIDATORS', undefined, '');
        this.validator = validateEmailFactory(rest);
    }
 
    validate(c: FormControl) {
        return this.validator(c);
    }
}
