import { FormControl } from '@angular/forms';
 
export class NipValidator {
    static valid(control: FormControl): any {
        let verificatorNip = new Array(6, 5, 7, 2, 3, 4, 5, 6, 7);
        let nip            = (!control.value || control.value === null ) ? '' : control.value.toString() || '';
        nip                = nip.replace(/\-/g, '');
        let message        = 'Błędny format nip.';
 
        if (nip !== '') {
            if (nip.length !== 10) {
                return {'incorrectNip': message};
            } else {
                let n = 0;
                for (let i = 0; i < 9; i++) {
                    n += nip[i] * verificatorNip[i];
                }
                n %= 11;
                if (n !== parseInt(nip[9], 10)) {
                    return {'incorrectNip': message};
                }
            }
        }
        return null;
    }
}
