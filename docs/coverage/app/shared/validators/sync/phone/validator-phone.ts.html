import { FormControl } from '@angular/forms';
 
export class PhoneValidator {
    static valid(control: FormControl): any {
        if (control.value) {
            let value = control.value.replace(/-|\.|\s/g, '');
            // let PHONE_REGEXP = /^[\+]?[(]?[0-9]{0,3}[)]?[-\s\.]?[0-9]{0,3}[-\s\.]?[0-9]{2,6}$/i;
            let PHONE_REGEXP = /^[\+]?[(]?[0-9]{0,3}[)]?[0-9]{2,12}$/i;
 
            if (value !== '' && !PHONE_REGEXP.test(value)) {
                return {'incorrectPhoneFormat': 'Błędny format nr. telefonu.'};
            }
        }
 
        return null;
    }
}
