import { Directive, forwardRef } from '@angular/core';
import { NG_VALIDATORS, FormControl } from '@angular/forms';
 
function validateOpeningHoursFactory() {
    return (c: FormControl) => {
        let EMAIL_REGEXP = /^[0-9]{1,2}:[0-9]{2,2}$/i;
 
        return EMAIL_REGEXP.test(c.value) ? null : {
            openingHours: {
                valid: false
            }
        };
    };
}
 
@Directive({
    selector: '[validateOpeningHours][ngModel],[validateOpeningHours][formControl]',
    providers: [
        { provide: NG_VALIDATORS, useExisting: forwardRef(() => OpeningHoursValidator), multi: true }
    ]
})
export class OpeningHoursValidator {
 
    validator: Function;
 
    constructor() {
        this.validator = validateOpeningHoursFactory();
    }
 
    validate(c: FormControl) {
        return this.validator(c);
    }
}
