import { Component, OnInit } from '@angular/core';
import { ClockService, DateService, DateFormat } from '../../shared';
 
@Component({
    selector: 'ecrm-clock-date',
    template: require('./clock-date.component.html'),
    styles: [require('./clock-date.component.scss')],
    providers: [ClockService, DateService]
})
export class ClockDateComponent implements OnInit {
    currentDate: String = '';
    currentTime: String = '';
    isClockEnable: Boolean = true;
    format: DateFormat = DateFormat.WEEKDAY_DAY__MONTH_YEAR;
    private counter: number = 1;
 
    constructor(private clock: ClockService, private date: DateService) {
        setTimeout(() => {
            this.runClock();
        }, 0);
    }
 
    runClock() {
        if (++this.counter === 5) {
            if (this.format === DateFormat.WEEKDAY_DAY__MONTH_YEAR) {
                this.format = DateFormat.YEARDAY_YEARWEEK;
            } else {
                this.format = DateFormat.WEEKDAY_DAY__MONTH_YEAR;
            }
            this.counter = 0;
        }
        let d = new Date();
        this.currentTime = this.clock.getTimeInFormat(d);
        this.currentDate = this.date.getDateInFormat(d, this.format);
        if (!this.isClockEnable) return;
        setTimeout(() => {
            this.runClock();
        }, 1000);
    }
 
 
 
    ngOnInit() { }
 
}
