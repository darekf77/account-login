import { Component, OnInit, Output, EventEmitter } from '@angular/core';
 
@Component({
    // moduleId: module.id,
    selector: 'ecrm-background-changer',
    template: require('./background-changer.component.html'),
    styles: [require('./background-changer.component.scss')]
})
export class BackgroundChangerComponent implements OnInit {
    backgrounds: string[] = [
        'assets/img/backgrounds/u0.png',
        'assets/img/backgrounds/luna.jpg'
    ];
    beingId: number = 0;
    @Output() backgroundChanged = new EventEmitter();
 
    constructor() {
        setTimeout(() => this.changeBackground());
 
    }
 
    changeBackground() {
        this.backgroundChanged.next(this.backgrounds[this.beingId]);
    }
 
    ngOnInit() { }
 
    previous() {
        if (this.beingId - 1 === -1) {
            this.beingId = this.backgrounds.length - 1;
            this.changeBackground();
            return;
        }
        this.beingId--;
        this.changeBackground();
    }
 
    next() {
        if (this.beingId + 1 === this.backgrounds.length) {
            this.beingId = 0;
            this.changeBackground();
            return;
        }
        this.beingId++;
        this.changeBackground();
    }
 
}
