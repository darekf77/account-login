import { Component, OnInit, ElementRef } from '@angular/core';
import { BackgroundChangerComponent } from './background-changer';
 
@Component({
    selector: 'ecrm-background',
    template: require('./background.component.html'),
    styles: [require('./background.component.scss')]
})
export class BackgroundComponent implements OnInit {
    constructor(private elementRef: ElementRef) { }
 
    ngOnInit() { }
 
    change(url) {
        let el = this.elementRef.nativeElement.children[0];
        el.style.backgroundImage = `url(${url})`;
        let classes = el.className;
        el.className = el.className + ' animated fadeIn';
        setTimeout(() => el.className = classes, 500);
    }
 
}
